sqlite Database smb_analytics_data(
  Schema default(
    Table industry(
      id:int*PK,
      industry_code:str*NN,
      industry_name:str*NN,
      default_rate_baseline:float*NN,
    )
    Table loan_officer(
      id:int*PK,
      employee_id:str*NN,
      first_name:str*NN,
      last_name:str*NN,
      email:str*NN,
      hire_date:date*NN,
      region:str*NN,
    )
    Table loan_status(
      id:int*PK,
      status_code:str*NN,
      status_name:str*NN,
      status_category:str*NN,
    )
    Table risk_grade(
      id:int*PK,
      grade_code:str*NN,
      grade_name:str*NN,
      min_credit_score:int*NN,
      max_credit_score:int*NN,
      interest_rate:float*NN,
      implied_default_rate:float*NN,
    )
    Table customer(
      id:int*PK,
      business_name:str*NN,
      tax_id:str*NN,
      industry_id:int*NN*FK->industry.id,
      state:str*NN,
      city:str*NN,
      founded_year:int*NN,
      annual_revenue:dec*NN,
      employee_count:int*NN,
      credit_score:int*NN,
      first_contact_date:date*NN,
      is_repeat_customer:bool*NN,
    )
    Table application(
      id:int*PK,
      application_number:str*NN,
      customer_id:int*NN*FK->customer.id,
      loan_officer_id:int*NN*FK->loan_officer.id,
      requested_amount:dec*NN,
      requested_term_months:int*NN,
      application_date:date*NN,
      decision_date:date,
      status_id:int*NN*FK->loan_status.id,
      rejection_reason:str,
    )
    Table loan(
      id:int*PK,
      loan_number:str*NN,
      application_id:int*NN*FK->application.id,
      customer_id:int*NN*FK->customer.id,
      risk_grade_id:int*NN*FK->risk_grade.id,
      approved_amount:dec*NN,
      interest_rate:float*NN,
      term_months:int*NN,
      monthly_payment:dec*NN,
      disbursement_date:date*NN,
      maturity_date:date*NN,
      current_status_id:int*NN*FK->loan_status.id,
      outstanding_balance:dec*NN,
    )
    Table default_event(
      id:int*PK,
      loan_id:int*NN*FK->loan.id,
      default_date:date*NN,
      installments_missed:int*NN,
      outstanding_at_default:dec*NN,
      recovery_amount:dec*NN,
      loss_amount:dec*NN,
      had_early_warning:bool*NN,
      warning_signals:str,
    )
    Table payment(
      id:int*PK,
      loan_id:int*NN*FK->loan.id,
      payment_date:date*NN,
      payment_amount:dec*NN,
      installment_number:int*NN,
      days_late:int*NN,
      payment_method:str*NN,
    )
    Table repayment_schedule(
      id:int*PK,
      loan_id:int*NN*FK->loan.id,
      installment_number:int*NN,
      due_date:date*NN,
      scheduled_payment:dec*NN,
      principal_portion:dec*NN,
      interest_portion:dec*NN,
      remaining_balance:dec*NN,
    )
  )
)
